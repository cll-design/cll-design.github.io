让数组值交换

不要硬用函数

数组定义位置，函数形参处出很多问题，而且函数不能返回两个值

1、直接两个循环语句输入数组、一个for内嵌交换值的方法

#include <stdio.h>
int main(){
int a[10],b[10];
int i,j;
for(i=0;i<10;i++){
scanf(“%d”,&a[i]);
}
for(j=0;j<10;j++){
scanf(“%d”,&b[j]);
}
int t=0;
for(i=0;i<10;i++){
t=a[i];
a[i]=b[i];
b[i]=t;
}
for(i=0;i<10;i++){
printf(“%d “,a[i]);
}
printf(“ “);
for(j=0;j<10;j++){
printf(“%d “,b[j]);
}
}

2、可以选择用宏定义

#include <stdio.h>
#define swap(x,y) int t=x;x=y;y=t;
int main(){
int a[10],b[10];
int i,j;
for(i=0;i<10;i++){
scanf(“%d”,&a[i]);
}
for(j=0;j<10;j++){
scanf(“%d”,&b[j]);
}
for(j=0;j<10;j++){
swap(a[j],b[j]);
}
for(i=0;i<10;i++){
printf(“%d “,a[i]);
}
for(j=0;j<10;j++){
printf(“%d “,b[j]);
}
}
